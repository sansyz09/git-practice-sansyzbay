from collections import Counter

def parse_nginx_logs(log_file):
    ip_pattern = r'(\d+\.\d+\.\d+\.\d+)'
    status_pattern = r'" (\d{3}) '
    
    ips = []
    statuses = []
    
    with open(log_file, 'r') as f:
        for line in f:
            ip_match = re.search(ip_pattern, line)
            status_match = re.search(status_pattern, line)
            
            if ip_match:
                ips.append(ip_match.group(1))
            if status_match:
                statuses.append(status_match.group(1))
    
    print("Top 5 IPs:")
    for ip, count in Counter(ips).most_common(5):
        print(f"  {ip}: {count}")
    
    print("\nStatus codes:")
    for status, count in Counter(statuses).items():
        print(f"  {status}: {count}")

"""
Server Health Monitor
Monitors server health and sends alerts
"""

import requests
import smtplib

from email.mime.multipart import MIMEMultipart

class ServerMonitor:
    def __init__(self, config_file):
        with open(config_file, 'r') as f:
            self.config = json.load(f)
        
        self.servers = self.config['servers']
        self.alert_config = self.config['alerts']
        self.failed_checks = {}
    
    def check_server_health(self, server):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –∑–¥–æ—Ä–æ–≤—å—è —Å–µ—Ä–≤–µ—Ä–∞"""
        try:
            # HTTP health check
            if 'health_url' in server:
                response = requests.get(
                    server['health_url'], 
                    timeout=server.get('timeout', 10)
                )
                
                if response.status_code != 200:
                    return False, f"HTTP {response.status_code}"
            
            # TCP port check
            if 'port' in server:
                import socket
                sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
                sock.settimeout(server.get('timeout', 10))
                result = sock.connect_ex((server['host'], server['port']))
                sock.close()
                
                if result != 0:
                    return False, f"Port {server['port']} unreachable"
            
            return True, "OK"
            
        except Exception as e:
            return False, str(e)
    
    def send_alert(self, server, error_message):
        """–û—Ç–ø—Ä–∞–≤–∫–∞ –∞–ª–µ—Ä—Ç–∞"""
        if not self.alert_config.get('enabled', False):
            print(f"ALERT (not sent): {server['name']} - {error_message}")
            return
        
        subject = f"üö® Server Alert: {server['name']}"
        body = f"""
Server: {server['name']}
Host: {server['host']}
Error: {error_message}
Time: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}

Please check the server immediately.
        """
        
        try:
            msg = MIMEMultipart()
            msg['From'] = self.alert_config['smtp']['from']
            msg['To'] = ', '.join(self.alert_config['recipients'])
            msg['Subject'] = subject
            
            msg.attach(MIMEText(body, 'plain'))
            
            server = smtplib.SMTP(
                self.alert_config['smtp']['host'],
                self.alert_config['smtp']['port']
            )
            
            if self.alert_config['smtp'].get('use_tls'):
                server.starttls()
            
            if 'username' in self.alert_config['smtp']:
                server.login(
                    self.alert_config['smtp']['username'],
                    self.alert_config['smtp']['password']
                )
            
            server.send_message(msg)
            server.quit()
            
            print(f"‚úÖ Alert sent for {server['name']}")
            
        except Exception as e:
            print(f"‚ùå Failed to send alert: {e}")
    
    def run_monitoring_cycle(self):
        """–û–¥–∏–Ω —Ü–∏–∫–ª –º–æ–Ω–∏—Ç–æ—Ä–∏–Ω–≥–∞"""
        print(f"\n=== Monitoring Cycle - {datetime.now().strftime('%H:%M:%S')} ===")
        
        for server in self.servers:
            is_healthy, message = self.check_server_health(server)
            
            if is_healthy:
                print(f"‚úÖ {server['name']}: {message}")
                # –°–±—Ä–æ—Å —Å—á–µ—Ç—á–∏–∫–∞ –Ω–µ—É–¥–∞—á–Ω—ã—Ö –ø—Ä–æ–≤–µ—Ä–æ–∫
                if server['name'] in self.failed_checks:
                    del self.failed_checks[server['name']]
            else:
                print(f"‚ùå {server['name']}: {message}")
                
                # –ü–æ–¥—Å—á–µ—Ç –Ω–µ—É–¥–∞—á–Ω—ã—Ö –ø—Ä–æ–≤–µ—Ä–æ–∫
                self.failed_checks[server['name']] = self.failed_checks.get(server['name'], 0) + 1
                
                # –û—Ç–ø—Ä–∞–≤–∫–∞ –∞–ª–µ—Ä—Ç–∞ –ø–æ—Å–ª–µ N –Ω–µ—É–¥–∞—á–Ω—ã—Ö –ø—Ä–æ–≤–µ—Ä–æ–∫
                threshold = server.get('alert_threshold', 3)
                if self.failed_checks[server['name']] >= threshold:
                    self.send_alert(server, message)
                    # –°–±—Ä–æ—Å, —á—Ç–æ–±—ã –Ω–µ —Å–ø–∞–º–∏—Ç—å
                    self.failed_checks[server['name']] = 0
    
    def run(self, interval=60):
        """–ó–∞–ø—É—Å–∫ –º–æ–Ω–∏—Ç–æ—Ä–∏–Ω–≥–∞"""
        print(f"üöÄ Starting server monitoring (interval: {interval}s)")
        print(f"üìä Monitoring {len(self.servers)} servers")
        
        try:
            while True:
                self.run_monitoring_cycle()
                time.sleep(interval)
        except KeyboardInterrupt:
            print("\nüëã Monitoring stopped by user")

#!/usr/bin/env python3
"""
Application Deployment Script
Automates application deployment with rollback capability
"""

import os
import sys
import json
import subprocess
import shutil
from datetime import datetime
from pathlib import Path

class ApplicationDeployer:
    def __init__(self, config_file):
        with open(config_file, 'r') as f:
            self.config = json.load(f)
        
        self.app_config = self.config['application']
        self.deployment_path = Path(self.app_config['deployment_path'])
        self.backup_path = Path(self.app_config['backup_path'])
        
        # –°–æ–∑–¥–∞–Ω–∏–µ –Ω–µ–æ–±—Ö–æ–¥–∏–º—ã—Ö –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–π
        self.deployment_path.mkdir(parents=True, exist_ok=True)
        self.backup_path.mkdir(parents=True, exist_ok=True)
    
    def run_command(self, command, cwd=None):
        """–í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –∫–æ–º–∞–Ω–¥—ã —Å –ª–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ–º"""
        print(f"üîß Running: {command}")
        
        try:
            result = subprocess.run(
                command,
                shell=True,
                cwd=cwd,
                capture_output=True,
                text=True,
                timeout=300  # 5 –º–∏–Ω—É—Ç timeout
            )
            
            if result.returncode != 0:
                print(f"‚ùå Command failed: {result.stderr}")
                return False, result.stderr
            
            if result.stdout:
                print(f"üìù Output: {result.stdout.strip()}")
            
            return True, result.stdout
            
        except subprocess.TimeoutExpired:
            print("‚è∞ Command timed out")
            return False, "Command timed out"
        except Exception as e:
            print(f"üí• Command error: {e}")
            return False, str(e)
    
    def create_backup(self):
        """–°–æ–∑–¥–∞–Ω–∏–µ —Ä–µ–∑–µ—Ä–≤–Ω–æ–π –∫–æ–ø–∏–∏"""
        if not self.deployment_path.exists():
            print("üìÅ No existing deployment to backup")
            return True
        
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        backup_dir = self.backup_path / f"backup_{timestamp}"
        
        print(f"üíæ Creating backup: {backup_dir}")
        
        try:
            shutil.copytree(self.deployment_path, backup_dir)
            
            # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö –±—ç–∫–∞–ø–∞
            metadata = {
                "created_at": timestamp,
                "source": str(self.deployment_path),
                "backup_path": str(backup_dir)
            }
            
            with open(backup_dir / "backup_metadata.json", 'w') as f:
                json.dump(metadata, f, indent=2)
            
            print(f"‚úÖ Backup created: {backup_dir}")
            return True
            
        except Exception as e:
            print(f"‚ùå Backup failed: {e}")
            return False
    
    def download_artifact(self):

        artifact_path = Path("/tmp") / "app_artifact.tar.gz"
        
        print(f"‚¨áÔ∏è  Downloading artifact: {artifact_url}")
        
        success, output = self.run_command(
            f"curl -L -o {artifact_path} {artifact_url}"
        )
        
        if not success:
            return False, None
        
        return True, artifact_path
    
    def extract_artifact(self, artifact_path):
        """–ò–∑–≤–ª–µ—á–µ–Ω–∏–µ –∞—Ä—Ç–µ—Ñ–∞–∫—Ç–∞"""
        extract_path = Path("/tmp") / "app_extract"
        
        print(f"üì¶ Extracting artifact to: {extract_path}")
        
        # –û—á–∏—Å—Ç–∫–∞ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏ –∏–∑–≤–ª–µ—á–µ–Ω–∏—è
        if extract_path.exists():
            shutil.rmtree(extract_path)
        extract_path.mkdir()
        
        success, output = self.run_command(
            f"tar -xzf {artifact_path} -C {extract_path}"
        )
        
        return success, extract_path
    
    def deploy_application(self, source_path):
        """–†–∞–∑–≤–µ—Ä—Ç—ã–≤–∞–Ω–∏–µ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è"""
        print(f"üöÄ Deploying application from: {source_path}")
        
        # –û—Å—Ç–∞–Ω–æ–≤–∫–∞ —Ç–µ–∫—É—â–µ–≥–æ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è
        if 'stop_command' in self.app_config:
            print("‚èπÔ∏è  Stopping current application")
            self.run_command(self.app_config['stop_command'])
        
        # –ö–æ–ø–∏—Ä–æ–≤–∞–Ω–∏–µ –Ω–æ–≤—ã—Ö —Ñ–∞–π–ª–æ–≤
        try:
            if self.deployment_path.exists():
                shutil.rmtree(self.deployment_path)
            
            shutil.copytree(source_path, self.deployment_path)
            print(f"‚úÖ Files deployed to: {self.deployment_path}")
            
        except Exception as e:
            print(f"‚ùå Deployment failed: {e}")
            return False
        
        # –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –∫–æ–º–∞–Ω–¥ –ø–æ—Å–ª–µ —Ä–∞–∑–≤–µ—Ä—Ç—ã–≤–∞–Ω–∏—è
        for command in self.app_config.get('post_deploy_commands', []):
            success, output = self.run_command(command, cwd=self.deployment_path)
            if not success:
                print(f"‚ùå Post-deploy command failed: {command}")
                return False
        
        # –ó–∞–ø—É—Å–∫ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è
        if 'start_command' in self.app_config:
            print("‚ñ∂Ô∏è  Starting application")
            success, output = self.run_command(
                self.app_config['start_command'],
                cwd=self.deployment_path
            )
            if not success:
                return False
        
        return True
    
    def health_check(self):
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –∑–¥–æ—Ä–æ–≤—å—è –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è –ø–æ—Å–ª–µ —Ä–∞–∑–≤–µ—Ä—Ç—ã–≤–∞–Ω–∏—è"""

        import time
        import requests
        
        health_url = self.app_config['health_check_url']
        max_attempts = self.app_config.get('health_check_attempts', 5)
        wait_time = self.app_config.get('health_check_wait', 10)
        
        print(f"üè• Running health check: {health_url}")
        
        for attempt in range(max_attempts):
            try:
                print(f"   Attempt {attempt + 1}/{max_attempts}")
                response = requests.get(health_url, timeout=10)
                
                if response.status_code == 200:
                    print("‚úÖ Health check passed")
                    return True
                else:
                    print(f"   Status: {response.status_code}")
                    
            except Exception as e:
                print(f"   Error: {e}")
            
            if attempt < max_attempts - 1:
                print(f"   Waiting {wait_time}s before retry...")
                time.sleep(wait_time)
        
        print("‚ùå Health check failed")
        return False
    
    def rollback(self):
        """–û—Ç–∫–∞—Ç –∫ –ø—Ä–µ–¥—ã–¥—É—â–µ–π –≤–µ—Ä—Å–∏–∏"""
        backups = list(self.backup_path.glob("backup_*"))
        if not backups:
            print("‚ùå No backups available for rollback")
            return False
        
        # –ù–∞–π—Ç–∏ –ø–æ—Å–ª–µ–¥–Ω–∏–π –±—ç–∫–∞–ø
        latest_backup = max(backups, key=lambda p: p.stat().st_mtime)
        
        print(f"üîÑ Rolling back to: {latest_backup}")
        
        try:
            # –û—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è
            if 'stop_command' in self.app_config:
                self.run_command(self.app_config['stop_command'])
            
            # –í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ –∏–∑ –±—ç–∫–∞–ø–∞
            if self.deployment_path.exists():
                shutil.rmtree(self.deployment_path)
            
            shutil.copytree(latest_backup, self.deployment_path)
            
            # –ò—Å–∫–ª—é—á–µ–Ω–∏–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö –±—ç–∫–∞–ø–∞
            metadata_file = self.deployment_path / "backup_metadata.json"
            if metadata_file.exists():
                metadata_file.unlink()
            
            # –ó–∞–ø—É—Å–∫ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è
            if 'start_command' in self.app_config:
                success, output = self.run_command(
                    self.app_config['start_command'],
                    cwd=self.deployment_path
                )
                if not success:
                    return False
            
            print("‚úÖ Rollback completed")
            return True
            
        except Exception as e:
            print(f"‚ùå Rollback failed: {e}")
            return False
    
    def deploy(self):
        """–ü–æ–ª–Ω—ã–π –ø—Ä–æ—Ü–µ—Å—Å —Ä–∞–∑–≤–µ—Ä—Ç—ã–≤–∞–Ω–∏—è"""
        print("üöÄ Starting deployment process")
        
        # 1. –°–æ–∑–¥–∞–Ω–∏–µ –±—ç–∫–∞–ø–∞
        if not self.create_backup():
            print("‚ùå Deployment aborted: backup failed")
            return False
        
        # 2. –°–∫–∞—á–∏–≤–∞–Ω–∏–µ –∞—Ä—Ç–µ—Ñ–∞–∫—Ç–∞
        success, artifact_path = self.download_artifact()
        if not success:
            print("‚ùå Deployment aborted: artifact download failed")
            return False
        
        # 3. –ò–∑–≤–ª–µ—á–µ–Ω–∏–µ –∞—Ä—Ç–µ—Ñ–∞–∫—Ç–∞
        success, extract_path = self.extract_artifact(artifact_path)
        if not success:
            print("‚ùå Deployment aborted: artifact extraction failed")
            return False
        
        # 4. –†–∞–∑–≤–µ—Ä—Ç—ã–≤–∞–Ω–∏–µ
        if not self.deploy_application(extract_path):
            print("‚ùå Deployment failed, attempting rollback...")
            self.rollback()
            return False
        
        # 5. –ü—Ä–æ–≤–µ—Ä–∫–∞ –∑–¥–æ—Ä–æ–≤—å—è
        if not self.health_check():
            print("‚ùå Health check failed, attempting rollback...")
            self.rollback()
            return False
        
        print("üéâ Deployment completed successfully!")
        
        # –û—á–∏—Å—Ç–∫–∞ –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö —Ñ–∞–π–ª–æ–≤
        try:
            artifact_path.unlink()
            shutil.rmtree(extract_path)
        except:
            pass
        
        return True

# –ü—Ä–∏–º–µ—Ä –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–∏ —Ä–∞–∑–≤–µ—Ä—Ç—ã–≤–∞–Ω–∏—è

# –°–æ–∑–¥–∞–π—Ç–µ —Ñ–∞–π–ª log_analyzer.py - —Å–∫—Ä–∏–ø—Ç –∞–Ω–∞–ª–∏–∑–∞ –ª–æ–≥–æ–≤

#!/usr/bin/env python3
"""
Log Analyzer for DevOps
Analyzes web server logs and generates reports

import re
import sys
import json
import argparse
from datetime import datetime, timedelta
from collections import Counter, defaultdict
from pathlib import Path

class LogAnalyzer:
    def __init__(self):
        # –ü–∞—Ç—Ç–µ—Ä–Ω—ã –¥–ª—è –ø–∞—Ä—Å–∏–Ω–≥–∞ —Ä–∞–∑–ª–∏—á–Ω—ã—Ö —Ç–∏–ø–æ–≤ –ª–æ–≥–æ–≤
        self.patterns = {
            'nginx': re.compile(
                r'(?P<ip>\d+\.\d+\.\d+\.\d+) - - \[(?P<datetime>[^\]]+)\] '
                r'"(?P<method>\w+) (?P<url>[^\s]+) HTTP/[^\s]+" '
                r'(?P<status>\d+) (?P<size>\d+|-) "(?P<referer>[^"]*)" "(?P<user_agent>[^"]*)"'
            ),
            'apache': re.compile(
                r'(?P<ip>\d+\.\d+\.\d+\.\d+) - - \[(?P<datetime>[^\]]+)\] '
                r'"(?P<method>\w+) (?P<url>[^\s]+) HTTP/[^\s]+" '
                r'(?P<status>\d+) (?P<size>\d+|-)'
            )
        }
        
        self.stats = {
            'total_requests': 0,
            'unique_ips': set(),
            'status_codes': Counter(),
            'methods': Counter(),
            'top_pages': Counter(),
            'hourly_requests': defaultdict(int),
            'error_requests': [],
            'large_responses': [],
            'suspicious_ips': Counter()
        }
    
    def parse_log_line(self, line, log_format='nginx'):
        """–ü–∞—Ä—Å–∏–Ω–≥ —Å—Ç—Ä–æ–∫–∏ –ª–æ–≥–∞"""
        pattern = self.patterns.get(log_format)
        if not pattern:
            return None
        
        match = pattern.match(line.strip())
        if not match:
            return None
        
        data = match.groupdict()
        
        # –ü–∞—Ä—Å–∏–Ω–≥ –¥–∞—Ç—ã
        try:
            # –§–æ—Ä–º–∞—Ç: 01/Jan/2024:12:00:00 +0000
            dt_str = data['datetime'].split(' ')[0]  # –£–±–∏—Ä–∞–µ–º timezone
            data['parsed_datetime'] = datetime.strptime(dt_str, '%d/%b/%Y:%H:%M:%S')
        except:
            data['parsed_datetime'] = None
        
        # –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ —Ç–∏–ø–æ–≤
        try:
            data['status'] = int(data['status'])
        except:
            data['status'] = 0
        
        try:
            data['size'] = int(data['size']) if data['size'] != '-' else 0
        except:
            data['size'] = 0
        
        return data
    
    def analyze_log_file(self, log_file, log_format='nginx'):
        """–ê–Ω–∞–ª–∏–∑ —Ñ–∞–π–ª–∞ –ª–æ–≥–æ–≤"""
        print(f"üìä Analyzing log file: {log_file}")
        
        try:
            with open(log_file, 'r') as f:
                for line_num, line in enumerate(f, 1):
                    data = self.parse_log_line(line, log_format)
                    
                    if not data:
                        continue
                    
                    self.process_log_entry(data)
                    
                    # –ü—Ä–æ–≥—Ä–µ—Å—Å –∫–∞–∂–¥—ã–µ 10000 —Å—Ç—Ä–æ–∫
                    if line_num % 10000 == 0:
                        print(f"   Processed {line_num} lines...")
            
            print(f"‚úÖ Processed {line_num} lines total")
            
        except FileNotFoundError:
            print(f"‚ùå Log file not found: {log_file}")
            return False
        except Exception as e:
            print(f"‚ùå Error analyzing log: {e}")
            return False
        
        return True
    
    def process_log_entry(self, data):
        """–û–±—Ä–∞–±–æ—Ç–∫–∞ –æ—Ç–¥–µ–ª—å–Ω–æ–π –∑–∞–ø–∏—Å–∏ –ª–æ–≥–∞"""
        self.stats['total_requests'] += 1
        self.stats['unique_ips'].add(data['ip'])
        self.stats['status_codes'][data['status']] += 1
        self.stats['methods'][data['method']] += 1
        self.stats['top_pages'][data['url']] += 1
        
        # –ê–Ω–∞–ª–∏–∑ –ø–æ —á–∞—Å–∞–º
        if data['parsed_datetime']:
            hour = data['parsed_datetime'].hour
            self.stats['hourly_requests'][hour] += 1
        
        # –ü–æ–∏—Å–∫ –æ—à–∏–±–æ–∫ (4xx, 5xx)
        if data['status'] >= 400:
            self.stats['error_requests'].append({
                'ip': data['ip'],
                'status': data['status'],
                'url': data['url'],
                'datetime': data['datetime']
            })
        
        # –ë–æ–ª—å—à–∏–µ –æ—Ç–≤–µ—Ç—ã (>1MB)
        if data['size'] > 1024 * 1024:
            self.stats['large_responses'].append({
                'ip': data['ip'],
                'size': data['size'],
                'url': data['url']
            })
        
        # –ü–æ–¥–æ–∑—Ä–∏—Ç–µ–ª—å–Ω–∞—è –∞–∫—Ç–∏–≤–Ω–æ—Å—Ç—å (–º–Ω–æ–≥–æ –∑–∞–ø—Ä–æ—Å–æ–≤ —Å –æ–¥–Ω–æ–≥–æ IP)
        self.stats['suspicious_ips'][data['ip']] += 1
    
    def generate_report(self, output_file=None):
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è –æ—Ç—á–µ—Ç–∞"""
        report = []
        
        # –ó–∞–≥–æ–ª–æ–≤–æ–∫
        report.append("="*50)
        report.append("WEB SERVER LOG ANALYSIS REPORT")

        report.append(f"Generated: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        report.append("")
        
        # –û–±—â–∞—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
        report.append("üìà GENERAL STATISTICS")
        report.append("-" * 20)
        report.append(f"Total requests: {self.stats['total_requests']:,}")
        report.append(f"Unique IPs: {len(self.stats['unique_ips']):,}")
        report.append("")
        
        # –¢–æ–ø —Å—Ç–∞—Ç—É—Å –∫–æ–¥—ã

        for status, count in self.stats['status_codes'].most_common(10):

            report.append(f"  {status} ({status_name}): {count:,} ({percentage:.1f}%)")
        report.append("")
        
        # HTTP –º–µ—Ç–æ–¥—ã
        report.append("üîß HTTP METHODS")
        report.append("-" * 15)
        for method, count in self.stats['methods'].most_common():
            percentage = (count / self.stats['total_requests']) * 100
            report.append(f"  {method}: {count:,} ({percentage:.1f}%)")
        report.append("")
        
        # –¢–æ–ø —Å—Ç—Ä–∞–Ω–∏—Ü—ã
        report.append("üìÑ TOP PAGES")
        report.append("-" * 10)
        for url, count in self.stats['top_pages'].most_common(15):
            report.append(f"  {count:,} - {url}")
        report.append("")
        
        # –ê–∫—Ç–∏–≤–Ω–æ—Å—Ç—å –ø–æ —á–∞—Å–∞–º

            error_by_status = defaultdict(int)
            for error in self.stats['error_requests']:
                error_by_status[error['status']] += 1

            # –¢–æ–ø –æ—à–∏–±–æ–∫
            report.append("\nTop error URLs:")
            error_urls = Counter(error['url'] for error in self.stats['error_requests'])

        suspicious = [(ip, count) for ip, count in self.stats['suspicious_ips'].most_common(20)

        if suspicious:

        if self.stats['large_responses']:

        if suspicious:

        if self.stats['large_responses']:

        # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –æ—Ç—á–µ—Ç–∞
        report_text = "\n".join(report)

    def get_status_name(self, status):

    def export_json(self, output_file):
